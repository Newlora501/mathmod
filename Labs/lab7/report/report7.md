–ú–æ–¥–µ–ª—å —Ä–µ–∫–ª–∞–º–Ω–æ–π –∫–∞–º–ø–∞–Ω–∏–∏ –æ–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è —Å–ª–µ–¥—É—é—â–∏–º–∏ –≤–µ–ª–∏—á–∏–Ω–∞–º–∏. –°—á–∏—Ç–∞–µ–º, —á—Ç–æ dn/dt - —Å–∫–æ—Ä–æ—Å—Ç—å –∏–∑–º–µ–Ω–µ–Ω–∏—è —Å–æ –≤—Ä–µ–º–µ–Ω–µ–º —á–∏—Å–ª–∞ –ø–æ—Ç—Ä–µ–±–∏—Ç–µ–ª–µ–π, —É–∑–Ω–∞–≤—à–∏—Ö –æ —Ç–æ–≤–∞—Ä–µ –∏ –≥–æ—Ç–æ–≤—ã—Ö –µ–≥–æ –∫—É–ø–∏—Ç—å, t - –≤—Ä–µ–º—è, –ø—Ä–æ—à–µ–¥—à–µ–µ —Å –Ω–∞—á–∞–ª–∞ —Ä–µ–∫–ª–∞–º–Ω–æ–π –∫–∞–º–ø–∞–Ω–∏–∏, n(t) - —á–∏—Å–ª–æ —É–∂–µ –∏–Ω—Ñ–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –∫–ª–∏–µ–Ω—Ç–æ–≤. –≠—Ç–∞ –≤–µ–ª–∏—á–∏–Ω–∞ –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–∞ —á–∏—Å–ª—É –ø–æ–∫—É–ø–∞—Ç–µ–ª–µ–π, –µ—â–µ –Ω–µ –∑–Ω–∞—é—â–∏—Ö –æ –Ω–µ–º, —ç—Ç–æ –æ–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è —Å–ª–µ–¥—É—é—â–∏–º –æ–±—Ä–∞–∑–æ–º: $$ \alpha_1(t)(N-n(t)) $$ –≥–¥–µ N - –æ–±—â–µ–µ —á–∏—Å–ª–æ –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω—ã—Ö –ø–ª–∞—Ç–µ–∂–µ—Å–ø–æ—Å–æ–±–Ω—ã—Ö –ø–æ–∫—É–ø–∞—Ç–µ–ª–µ–π, $$ \alpha_1(t)>0 $$ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É–µ—Ç –∏–Ω—Ç–µ–Ω—Å–∏–≤–Ω–æ—Å—Ç—å —Ä–µ–∫–ª–∞–º–Ω–æ–π –∫–∞–º–ø–∞–Ω–∏–∏ (–∑–∞–≤–∏—Å–∏—Ç –æ—Ç –∑–∞—Ç—Ä–∞—Ç –Ω–∞ —Ä–µ–∫–ª–∞–º—É –≤ –¥–∞–Ω–Ω—ã–π –º–æ–º–µ–Ω—Ç –≤—Ä–µ–º–µ–Ω–∏). –ü–æ–º–∏–º–æ —ç—Ç–æ–≥–æ, —É–∑–Ω–∞–≤—à–∏–µ –æ —Ç–æ–≤–∞—Ä–µ –ø–æ—Ç—Ä–µ–±–∏—Ç–µ–ª–∏ —Ç–∞–∫–∂–µ —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω—è—é—Ç –ø–æ–ª—É—á–µ–Ω–Ω—É—é –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é —Å—Ä–µ–¥–∏ –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω—ã—Ö –ø–æ–∫—É–ø–∞—Ç–µ–ª–µ–π, –Ω–µ –∑–Ω–∞—é—â–∏—Ö –æ –Ω–µ–º (–≤ —ç—Ç–æ–º —Å–ª—É—á–∞–µ —Ä–∞–±–æ—Ç–∞–µ—Ç —Ç.–Ω. —Å–∞—Ä–∞—Ñ–∞–Ω–Ω–æ–µ —Ä–∞–¥–∏–æ). –≠—Ç–æ—Ç –≤–∫–ª–∞–¥ –≤ —Ä–µ–∫–ª–∞–º—É –æ–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è –≤–µ–ª–∏—á–∏–Ω–æ–π $$ \alpha_2(t)n(t)(N-n(t)) $$ —ç—Ç–∞ –≤–µ–ª–∏—á–∏–Ω–∞ —É–≤–µ–ª–∏—á–∏–≤–∞–µ—Ç—Å—è —Å —É–≤–µ–ª–∏—á–µ–Ω–∏–µ–º –ø–æ—Ç—Ä–µ–±–∏—Ç–µ–ª–µ–π —É–∑–Ω–∞–≤—à–∏—Ö –æ —Ç–æ–≤–∞—Ä–µ. –ú–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–∞—è –º–æ–¥–µ–ª—å —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏—è —Ä–µ–∫–ª–∞–º—ã –æ–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è —É—Ä–∞–≤–Ω–µ–Ω–∏–µ–º: $$ \frac{\partial n}{\partial t}=(\alpha_1(t)+\alpha_2(t)n(t))(N-n(t)) $$

using DifferentialEquations

"–∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —É—Ä–∞–≤–Ω–µ–Ω–∏—è 1"
a1 = 0.64
a2 = 0.00004

c1 = 0.4
c2 = 0.3
N = 1403    
n0 = 9


function  odn_f(du, u, p, t)
    x, y, z = u
    du[1]= (a1+a2*u[1])*(N-u[1])                 
    du[2]= (b1+b2*u[1])*(N-u[1])                     
    du[3]= (c1*t+c2*sin(2*t)*u[1])*(N-u[1])           
end
u0 = [n0, n0, n0]
tspan =(0.0, 30.0)
prob1 = ODEProblem(odn_f, u0, tspan)
sol1 = solve(prob1, dtmax=0.1)

N1 = [u[1] for u in sol1.u]
N2 = [u[2] for u in sol1.u]
N3 = [u[3] for u in sol1.u]
T = [t for t in sol1.t]

plt = 
    plot(
        layout=(1,2),
